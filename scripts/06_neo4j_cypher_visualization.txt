// TCGA GBM Database Project
// Neo4j script for loading SQL-derived differentially expressed genes
//
// CSV source:
// https://raw.githubusercontent.com/arstephens/TCGA-Glioblastoma-Database/refs/heads/main/neo4j/differentially_expressed_genes.csv
//
// Graph structure:
// (:Gene)-[:HIGHER_IN]->(:Subtype)



// Load differentially expressed genes
LOAD CSV WITH HEADERS
FROM 'https://raw.githubusercontent.com/arstephens/TCGA-Glioblastoma-Database/refs/heads/main/neo4j/differentially_expressed_genes.csv' AS row

MERGE (g:Gene {gene_id: toInteger(row.gene_id)})

SET
    g.hugo_symbol = row.hugo_symbol,
    g.entrez_gene_id = CASE
        WHEN row.entrez_gene_id IS NULL OR row.entrez_gene_id = '' THEN NULL
        ELSE toInteger(row.entrez_gene_id)
    END,
    g.mean_classical_expression = toFloat(row.mean_classical_expression),
    g.mean_proneural_expression = toFloat(row.mean_proneural_expression),
    g.classical_sample_count = toInteger(row.classical_sample_count),
    g.proneural_sample_count = toInteger(row.proneural_sample_count),
    g.log2_fold_change = toFloat(row.log2_fold_change),
    g.expression_direction = row.expression_direction

MERGE (classical:Subtype {name: 'Classical'})
MERGE (proneural:Subtype {name: 'Proneural'})

FOREACH (_ IN CASE WHEN row.expression_direction = 'Higher in Classical' THEN [1] ELSE [] END |
    MERGE (g)-[:HIGHER_IN {comparison: 'Classical vs Proneural'}]->(classical)
)

FOREACH (_ IN CASE WHEN row.expression_direction = 'Higher in Proneural' THEN [1] ELSE [] END |
    MERGE (g)-[:HIGHER_IN {comparison: 'Classical vs Proneural'}]->(proneural)
);


// Check gene counts by subtype
MATCH (g:Gene)-[:HIGHER_IN]->(s:Subtype)
RETURN
    s.name AS subtype,
    count(g) AS gene_count;


// Visualize 100 genes connected to subtype nodes
MATCH (g:Gene)-[r:HIGHER_IN]->(s:Subtype)
RETURN g, r, s
LIMIT 100;


// Visualize a balanced graph:
// top 25 genes higher in Classical and top 25 genes higher in Proneural
MATCH (g:Gene)-[r:HIGHER_IN]->(s:Subtype {name: 'Classical'})
WITH g, r, s
ORDER BY g.log2_fold_change DESC
LIMIT 25
RETURN g, r, s

UNION

MATCH (g:Gene)-[r:HIGHER_IN]->(s:Subtype {name: 'Proneural'})
WITH g, r, s
ORDER BY g.log2_fold_change ASC
LIMIT 25
RETURN g, r, s;
